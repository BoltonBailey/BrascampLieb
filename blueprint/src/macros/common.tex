% In this file you should put all LaTeX macros and settings to be used both by
% the pdf version and the web version.
% This should be most of your macros.

% The theorem-like environments defined below are those that appear by default
% in the dependency graph. See the README of leanblueprint if you need help to
% customize this.
% The configuration below use the theorem counter for all those environments
% (this is what the [theorem] arguments mean) and never resets it.
% If you want for instance to number them within chapters then you can add
% [chapter] at the end of the next line.
\newtheorem{theorem}{Theorem}[chapter]
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{corollary}[theorem]{Corollary}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}

\newcommand{\set}[1]{\left\{ #1 \right\}} % macro for set builder notation
\newcommand{\abs}[1]{\left\lvert#1\right\rvert}    % macro for absolute value natation
\newcommand{\norm}[1]{\left\lVert #1 \right\rVert} % macro for norm notation

\newcommand{\N}{\mathbb{N}} % natural numbers
\newcommand{\Z}{\mathbb{Z}} % integers
\newcommand{\Q}{\mathbb{Q}} % rational numbers
\newcommand{\R}{\mathbb{R}} % real numbers
\newcommand{\C}{\mathbb{C}} % complex numbers

\newcommand{\eps}{\varepsilon} % "small positive number" :)

\newcommand{\PR}{\mathsf{P}} % probability measure
\newcommand{\EV}{\mathsf{E}} % expected value
\newcommand{\PRspace}{\Omega} % sample space
\newcommand{\PRsigma}{\mathscr{F}} % sigma algebra of events
\newcommand{\indicator}{\mathbf{1}} % indicator function (or indicator random variable)
\newcommand{\indicOf}[1]{\indicator_{{#1}}} % indicator of a set (or of an event)
\newcommand{\tendsas}{\overset{\mathrm{a.s.}}{\longrightarrow}} % convergence a.s.
\newcommand{\tendspr}{\overset{\PR}{\longrightarrow}} % convergence in probability
\newcommand{\tendslaw}{\overset{\mathrm{law}}{\longrightarrow}} % convergence in distribution (law)
\newcommand{\tendsweakly}{\overset{\mathrm{w}}{\longrightarrow}} % weak convergence (of measures)

\newcommand{\genspace}{\mathcal{X}} % a generic space
\newcommand{\genspaceA}{\mathcal{X}} % a generic space A (when two or more are needed)
\newcommand{\genspaceB}{\mathcal{Y}} % a generic space B (when two or more are needed)
\newcommand{\gendist}{\mathsf{d}} % a generic distance (metric)
\newcommand{\gendistOf}[1]{\gendist_{{#1}}} % distance (metric) on a given space
\newcommand{\gendistA}{\gendistOf{\genspaceA}} % distance (metric) on space A
\newcommand{\gendistB}{\gendistOf{\genspaceB}} % distance (metric) on space B
\newcommand{\gensigma}{\mathscr{X}} % a generic sigma algebra
\newcommand{\borel}{\mathscr{B}} % Borel sigma algebra
\newcommand{\borelOf}[1]{\borel({#1})} % Borel sigma algebra on a given space

\newcommand{\closure}[1]{\overline{{#1}}} % closure of a set

\newcommand{\Leb}{\Lambda}%{\mu_{\mathrm{Leb}}}
